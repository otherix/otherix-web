From zero to a running VM

# 1. install the operator CLI
curl -fsSL get.otherix.dev | OTHERIX_COMPONENT=cli sh

# 2. install the control plane on a Linux host
curl -fsSL get.otherix.dev | OTHERIX_COMPONENT=api sudo -E sh

# 3. create your first VM from an image URL
otherix vm create --image-url https://example.com/noble.qcow2 \
  --arch amd64 --disk-gib 20
